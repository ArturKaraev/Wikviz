        private void button1_Click(object sender, EventArgs e)
        {
            TextAdress.Visible = false;  
            TextPort.Visible = false;
            LabelAdress.Visible = false;
            LabelPort.Visible = false;
            Button1.Visible = false;     

            Button2.Visible = true;      
            TextAnswer.Visible = true;
            LabelQuest.Visible = true;

            connection.Open();
            string query = "SELECT Question FROM QUESTIONS WHERE Id = 1";

            using (var command = new SqlCommand(query, connection))
            using (var reader = command.ExecuteReader())
            {
                if (reader.Read())
                {
                    LabelQuest.Text = reader["Question"].ToString();
                }
            }
            connection.Close();
            
        }

                        if (reader.Read())

                {

            }
            catch (Exception ex)
            {
                MessageBox.Show($"Database error: {ex.Message}"); // Лучше использовать MessageBox для WinForms
            }
            finally
            {
                if (connection.State == System.Data.ConnectionState.Open)
                {
                    connection.Close();
                }
            }
        }
